# Python Basics: If-Else Conditions

# Check student eligibility (criteria = 680)
marks = 680
if marks >= 680:
    print("Student is eligible")

# Check student eligibility (criteria = 650)
marks = 600
if marks >= 650:
    print("Student is eligible")
else:
    print("Student is not eligible")

# Performance grading
marks = 100

if marks >= 80:
    print("Excellent")
elif marks >= 60:
    print("Good")
elif marks >= 50:
    print("Average")
else:
    print("Fail")
